let font; let deepPurple = '#2E0854'; let black = '#000000'; let gamingBlue = '#00F0FF'; let neonPink = '#FF003C'; let grey = '#CCCCCC'; let white = '#FFFFFF'; let dinoX, dinoY; let targetX, targetY; let dinoYLimit; let clouds = []; let moon; let boxInfo = [ { name: 'FOOD' }, { name: 'MUSIC' }, { name: 'ATHLEISURE' } ]; function preload() { // Use a reliable pixelated font font = loadFont('https://fastly.jsdelivr.net/gh/google/fonts/ofl/pressstart2p/PressStart2P-Regular.ttf'); } function setup() { createCanvas(windowWidth, windowHeight); textFont(font); noCursor(); // Hide the default cursor for a full-page feel dinoX = width / 2; dinoY = height / 2; targetX = dinoX; targetY = dinoY; // Set up background elements and constraint area moon = { x: width * 0.9, y: height * 0.1, size: 80 }; dinoYLimit = height * 0.65; // Defined vertical limit for character follow for (let i = 0; i < 3; i++) { clouds.push({ x: width + random(100, 400), y: height * 0.15 + i * height * 0.1, size: random(1.5, 3), speed: random(0.2, 0.5) }); } } function draw() { background(deepPurple); // 1. Draw Night Sky and Moon (Deep Purple, Pixelated) drawMoon(); drawClouds(); // 2. Draw Interactive Character (Dinosaur with items) updateAndDrawDino(); // 3. Draw Text Hierarchy and Information drawInfoText(); // 4. Draw Interactive Option Boxes drawInteractiveBoxes(); // 5. Custom Pixelated Cursor drawPixelCursor(); } function drawMoon() { push(); noStroke(); fill(white); rect(moon.x, moon.y, moon.size, moon.size, 10); // Pixelated moon shape fill('#E0E0E0'); // Simple pixel craters rect(moon.x + moon.size * 0.2, moon.y + moon.size * 0.3, moon.size * 0.1, moon.size * 0.1, 2); rect(moon.x + moon.size * 0.5, moon.y + moon.size * 0.5, moon.size * 0.1, moon.size * 0.1, 2); rect(moon.x + moon.size * 0.7, moon.y + moon.size * 0.2, moon.size * 0.1, moon.size * 0.1, 2); pop(); } function drawClouds() { push(); noStroke(); fill(grey + '66'); // Semitransparent pixel clouds for (let cloud of clouds) { cloud.x -= cloud.speed; // Move from right to left if (cloud.x + cloud.size * 100 < 0) { cloud.x = width + random(50, 200); } // Simple pixelated cloud shape draw drawPixelCloud(cloud.x, cloud.y, cloud.size); } pop(); } function drawPixelCloud(x, y, scale) { push(); translate(x, y); scale(scale); rect(0, 0, 40, 20); rect(10, -10, 30, 20); rect(20, -20, 20, 20); rect(-10, -10, 10, 10); pop(); } function updateAndDrawDino() { targetX = mouseX; targetY = mouseY; // Constrain Y position to the specific interaction area let currentTargetY = min(targetY, dinoYLimit - 60); // Smoothly move the dinosaur dinoX = lerp(dinoX, targetX, 0.05); dinoY = lerp(dinoY, currentTargetY, 0.05); push(); translate(dinoX - 40, dinoY - 60); // Offset to draw from the feet/base // Draw Pixel Dino (Google style, but black) fill(black); noStroke(); rect(0, 0, 60, 40); // Body rect(10, -20, 40, 20); // Head fill(white); rect(20, -10, 10, 10); // Eye fill(black); rect(25, -5, 2, 2); // Pupil rect(10, -30, 5, 10); // Spikes rect(20, -35, 5, 10); rect(30, -30, 5, 10); rect(5, 30, 15, 30); // Feet rect(40, 30, 15, 30); // Laptop on back fill(grey); rect(10, -20, 25, 20); // Laptop base fill(white); rect(15, -15, 15, 10); // Screen fill(gamingBlue); rect(16, -14, 13, 8); // Screen display // Pencil in "hand" (small attachment) fill('#E1B07A'); // Light wood pencil color rect(50, 10, 5, 15); fill(black); rect(50, 25, 5, 2); // Pencil tip pop(); } function drawInfoText() { push(); // Optional subtle frame for the landing page group area noFill(); stroke(gamingBlue + '33'); rect(20, 20, width - 40, dinoYLimit - 20, 5); translate(100, 100); fill(black); // 1. Main Title (Huge) textSize(80); text("DATA VIZ", 0, 0); // 2. Subtitle 1 (Small) textSize(16); text("SEM 3 IxD, NID Bengaluru", 10, 120); // 3. Subtitle 2 (Medium, Pink) textSize(24); fill(neonPink); text("Mochi does is here", 10, 170); // 4. Description (Small) fill(black); textSize(18); let descText = "Just a girl slaying here data viz module ;}\n[ps- Crying in coding* ]"; textLeading(30); text(descText, 10, 220); pop(); } function drawInteractiveBoxes() { push(); let boxWidth = (width - 200) / 3; let boxHeight = height * 0.2; let boxY = height - boxHeight - 50; let gap = 20; for (let i = 0; i < boxInfo.length; i++) { let box = boxInfo[i]; let boxX = 100 + i * (boxWidth + gap); // Check for hover let isHovering = mouseX > boxX && mouseX < boxX + boxWidth && mouseY > boxY && mouseY < boxY + boxHeight; // Draw Box Border (Pixelated) strokeWeight(4); if (isHovering) { stroke(gamingBlue); fill(black); } else { stroke(gamingBlue + 'aa'); noFill(); } rect(boxX, boxY, boxWidth, boxHeight, 10); // Draw Crossed Arrow Icon push(); translate(boxX + 20, boxY + 20); scale(1.5); strokeWeight(3); if (isHovering) { stroke(white); } else { stroke(grey); } drawCrossedArrow(); pop(); // Draw Text Label noStroke(); if (isHovering) { fill(white); } else { fill(white + '99'); } textSize(18); textAlign(CENTER, CENTER); text(box.name, boxX + boxWidth / 2, boxY + boxHeight / 2); } pop(); } function drawCrossedArrow() { // Simple pixelated crossed arrows line(0, 0, 20, 20); line(0, 20, 20, 0); } function drawPixelCursor() { push(); noFill(); stroke(white); strokeWeight(2); rect(mouseX - 5, mouseY - 5, 10, 10); line(mouseX, mouseY - 5, mouseX, mouseY + 5); line(mouseX - 5, mouseY, mouseX + 5, mouseY); pop(); } function windowResized() { resizeCanvas(windowWidth, windowHeight); dinoYLimit = height * 0.65; // Update background elements moon.x = width * 0.9; moon.y = height * 0.1; for (let i = 0; i < clouds.length; i++) { clouds[i].x = width + random(100, 400); } }
